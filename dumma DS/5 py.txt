#(ii)count keyword from the soure file
import token
import tokenize
import keyword
s=open('f1.txt').readline
c=0
l=[]
for i in tokenize.generate_tokens(s):
    if i.type==token.NAME and keyword.iskeyword(i.string):
        c+=1
        l.append(i.string)
print(c)
print(l)
